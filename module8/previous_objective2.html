Objective 8.2 - Building the middleware to connect a back end database to a front end of a web application

In the previous page you created a MongoDB database called cars . What we want to work on now is creating the Node.js scripts that will provide the middleware that can connect this database to a web front end. The method that we will use involves proving some REST (Representational State Transfer) routes that are essentially URLs that can be used to carry out a certain process. A typical REST API (Application Peripheral Interface) should be able to carry out the basic CRUD (Create, Read, Update, Delete) operations on the database.

If we were working with a relational database, then Create would correspond to a SQL Insert query, Read would correspond to a SQL Select query, Update would correspond to a SQL Update query and Delete would correspond to a SQL Delete query.

Since we are working with MongoDB (a NoSQL database), CRUD will correspond to different commands from the SQL commands, but the functionality will be similar.

Creating a REST route

We will start by creating a very simple route that will just serve as a debugging tool to start. We are going to create a file called "index.js" that will act as the server script, that serves out the connection between the back end database and the front end web application. Just as with a web application using PHP/Apache and PostgreSQL, we develop from the back to the front.

We have already built our back end database, and are now working on the middleware. As much as possible, we want to build up the middleware completely, before we even start on the front end web application. The file "index.js" will be launched by Node.js to provide this middleware.

Setting up to update a car document

There are two processes that we need to take care of to update a car document. To see that there are two processes requires that you think ahead to the web front end. In order to edit existing data, you need to have a way of selecting a document and then displaying the current contents of that document. That will allow the user to change only the necessary parts of the document. The second process is to take the changed values for the document and update the database.

The new version of "queries.js" supports that first process of getting the existing data for a document: 


// queries.js
const mongoose = require("mongoose");
mongoose.connect("mongodb://172.17.0.2:27017/cars_db",
   {
      useNewUrlParser: true,
      useUnifiedTopology: true
   })
   .catch(error => {console.log(error)});
let db = mongoose.connection;
//db.close();
let carSchema = mongoose.Schema({
   make: String,
   model: String,
   price: Number   
});
let Car = mongoose.model('Car',carSchema);
function getCars(request, response) {
   Car.find().then(cars => {
      response.status(200).json(cars); 
   });
}
function createCar(request, response) {
   const {car_make, car_model, car_price} = request.body;
   let c1 = Car({
      make: car_make,
      model: car_model,
      price: car_price
   });
   c1.save(function(error, car) {
      if (error) {
         throw error;
      }
      response.status(201).send('Car created\n');

Updating a car document

Let's modify "queries.js" to add a function, updateCar() that can be used to update a car document. Here is the new version of "queries.js": 


// queries.js
const mongoose = require("mongoose");
mongoose.connect("mongodb://172.17.0.2:27017/cars_db",
   {
      useNewUrlParser: true,
      useUnifiedTopology: true
   })
   .catch(error => {console.log(error)});
let db = mongoose.connection;
//db.close();
let carSchema = mongoose.Schema({
   make: String,
   model: String,
   price: Number
});
let Car = mongoose.model('Car',carSchema);
function getCars(request, response) {
   Car.find().then(cars => {
      response.status(200).json(cars);
   });
}
function createCar(request, response) {
   const {car_make, car_model, car_price} = request.body;
   let c1 = Car({
      make: car_make,
      model: car_model,
      price: car_price
   });
   c1.save(function(error, car) {
      if (error) {
         throw error;
      }
      response.status(201).send('Car created\n');
   });
}
function getCarById(request, response) {
   const id = request.params.id;
   Car.findById(mongoose.mongo.ObjectId(id), (error, car) => {
      if (error) {
         throw error;
      }
      response.status(200).json(car);
   });
}
function updateCar(request, response) {
   const id = request.params.id;
   const {car_make, car_model, car_price} = request.body;
   Car.findByIdAndUpdate(id, {
         make: car_make,
         model: car_model,
         price: car_price
      },
      {
         useFindAndModify: false
      },
      (error, car) => {
         if (error) {
            throw error;
         }
         response.status(200).send('Car updated\n');
      }
   );
}
module.exports = { getCars, createCar, getCarById,
   updateCar };

A better updateCar() function

As it stands the updateCar() function is not the only way to define that function. The following code shows a different version of "queries.js" that has an updateCar() function that allows for updating only the keys you want when modifying a car document: 


// queries.js
const mongoose = require("mongoose");
mongoose.connect("mongodb://172.17.0.2:27017/cars_db",
   {
      useNewUrlParser: true,
      useUnifiedTopology: true
   })
   .catch(error => {console.log(error)});
let db = mongoose.connection;
//db.close();
let carSchema = mongoose.Schema({
   _id: mongoose.Schema.ObjectId,
   make: String,
   model: String,
   price: Number
});
let Car = mongoose.model('Car',carSchema);
function getCars(request, response) {
   Car.find().then(cars => {
      response.status(200).json(cars);
   });
}
function createCar(request, response) {
   const {make, model, price} = request.body;
   let c1 = Car({
      _id: mongoose.mongo.ObjectId(),
      make: make,
      model: model,

   const id = request.params.id;

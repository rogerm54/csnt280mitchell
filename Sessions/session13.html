drop view survey_questions_view cascade;
drop view question_answers_view cascade;

create view survey_questions_view as
   select surveys.id, surveys.title, surveys.survey_code,
   questions.question_text from surveys join
   questions on surveys.id=questions.survey_id;

create view question_answers_view as
   select questions.id, questions.survey_id, 
   questions.question_text,
   answers.num1, answers.num2, answers.num3,
   answers.num4, answers.num5 from questions
   join answers on questions.id=answers.question_id;
